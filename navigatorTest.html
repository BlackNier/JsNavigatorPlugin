    导航栏使用方法:
    <script src="js/Navigator.js" async></script>
    <!--引入js文件-->
    <link rel="stylesheet" href="css/style.css" type="text/css" />
    <!--引入css文件-->
    建立导航栏:
    <div class="nav">
    <div class="mainTitle">导航栏主标题</div>
    <li><a href="#">链接1</a></li>
    <li><a href="#">链接2</a></li>
    <div class="subTitle">导航栏副标题</div>
    <li><a href="#">链接3</a></li>
    </div>
    下拉菜单:
    <div class="dropdown">
    <li class="dropdown-Head">下拉菜单</li>
    <ul class="dropdown-Menu">
    <li><a href="#">菜单项1</a></li>
    <li><a href="#">菜单项2</a></li>
    <li><a href="#">菜单项3</a></li>
    </ul>
    </div>
    自定义导航栏宽度:
    <div data-width="220px" class="nav"</div>
    <!--定义px为单位的宽度-->
    <div data-width="15%" class="nav"</div>
    <!--定义%为单位的宽度-->
    添加图标:
    <div data-icon-title="home" class="mainTitle">导航栏主标题</div>
    <!--为标题添加图标-->
    <li data-icon="category"><a href="#">链接1</a></li>
    <!--为列表项添加图标-->
    图标样式演示:

    • data-icon="home"
